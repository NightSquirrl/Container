## 校验中文

```javascript
// 测试字符串
const testStrings = [
    "你好世界",           // ✓ 常用中文
    "𠀀𠀁𠀂",            // ✓ 扩展B区汉字
    "Hello",             // ✗ 英文
    "你好123",           // ✗ 包含数字
    "你好 world",        // ✗ 包含英文
];

// 使用 Unicode 属性转义 (最全面，推荐)
const chineseRegex = /^[\p{Script=Han}]+$/u;

testStrings.forEach(str => {
    console.log(`${str}: ${chineseRegex.test(str) ? '✓ 匹配' : '✗ 不匹配'}`);
});
```
